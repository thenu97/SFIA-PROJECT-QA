29def numeric_part(s): 

30 """Returns the leading numeric part of a string. 

31 

32 >>> numeric_part("20-alpha") 

33 20 

34 >>> numeric_part("foo") 

35 >>> numeric_part("16b") 

36 16 

37 """ 

38 

39 m = re_numeric_part.match(s) 

40 if m: 

41 return int(m.group(1)) 

42 return None

251 def literal(self, o): 

252 """If o is a single object, returns an SQL literal as a string. 

253 If o is a non-string sequence, the items of the sequence are 

254 converted and returned as a sequence. 

255 

256 Non-standard. For internal use; do not use this in your 

257 applications. 

258 """ 

259 if isinstance(o, unicode): 

260 s = self.string_literal(o.encode(self.encoding)) 

261 elif isinstance(o, bytearray): 

262 s = self._bytes_literal(o) 

263 elif isinstance(o, bytes): 

264 if PY2: 

265 s = self.string_literal(o) 

266 else: 

267 s = self._bytes_literal(o)

271 s = self.escape(o, self.encoders) 

272 if isinstance(s, unicode): 

273 s = s.encode(self.encoding) 

274 assert isinstance(s, bytes)
